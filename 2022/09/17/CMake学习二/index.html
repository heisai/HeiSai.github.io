CMake学习二

Cmake 进阶使用

添加子目录（子工程–可执行文件）

  • 指令

      将子目录添加到构建。

  add_subdirectory(source_dir [binary_dir] [EXCLUDE_FROM_ALL])

  将子目录添加到构建。source_dir 指定源 CMakeLists.txt 和代码文件所在的目录。如果它是相对路径，它
  将相对于当前目录进行评估（典型用法），但它也可能是绝对路径。binary_dir指定放置输出文件的目录

  • 自己理解

      将source_dir 目录生成的东西（可执行文件， 库）添加到指定的目录，
  可以搞多工程（一个主工程， 下面有多个子工程（可执行文件，库））

  • 使用案例：

      |-- CMakeLists.txt
  |-- src
  |   |-- CMakeLists.txt
  |   |-- MyWidget.cpp
  |   |-- MyWidget.h
  |   `-- main.cpp
  `-- test
      |-- CMakeLists.txt
      `-- main.cpp


  // 总目录：

  cmake_minimum_required(VERSION 3.5)

  project(HELLO LANGUAGES  C CXX)

  set(CMAKE_CXX_STANDARD 17)

  set(CMAKE_CXX_STANDARD_REQUIRED ON)

  ADD_SUBDIRECTORY(src)

  ADD_SUBDIRECTORY(test)

  =======================================================
  // src 目录

  file(GLOB_RECURSE SOURCE_FILES  ./*.cpp)                   # 添加 .cpp

  file(GLOB_RECURSE HEADER_FILES  ./*.h)                     # 添加 .h

  ADD_EXECUTABLE(hello main.cpp  ${HEADER_FILES} ${SOURCE_FILES})

  SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin3)

  ========================================================
  // test 目录 

  ADD_EXECUTABLE(test main.cpp)

  SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin3)

指定地方保存二进制文件

  • 方法一：通用情况

      # 将生成的可执行程序输出到 {build}/bin

  SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin)

  # 将生成的库文件输出到 {build}/bin

  SET(LIBRARY_OUTPUT_PATH ${PROJECT_BINARY_DIR}/lib)

  指最终生成的可执行程序 或者最终的共享库，不包含编译生成 的中间文件)

  • 方法二：特殊情况(有子目录嵌套)

      #将 src 目录下的二进制文件 输出到{build}/src 目录
  ADD_SUBDIRECTORY(src)
  #将 src 目录下的二进制文件 输出到{build}/test 目录
  ADD_SUBDIRECTORY(test)
